==html기본구조 (*.html, *.htm)
태그:요소를 만들때 사용하는 기호,
    -시작태그와 끝태그가 존재 (<h1>내용  </h1>)
    -단독으로 사용하는 태그
    html5   <br>
    xhtml   <br/>   <--개발자들이 선호
---
<!DOCTYPE html>    <!-- html5 임을 브라우져에게 알림 -->
<html> <!-- html의 루트요소, lang 속성 입력가능 -->
    <head>  <!-- body에 필요한 css3 또는 자바스크립트등 제공 -->
        <title>html</title>
    </head>
    <body><!-- 사용자에게 실제보여지는 부분 -->
    first homepage.
    </body>
</html>
